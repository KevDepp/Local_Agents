L’idée la plus facile et la plus robuste, c’est d’éviter au maximum le “clic/coordonnées souris” (cassant dès que l’UI bouge) et de s’appuyer sur des raccourcis clavier stables dans chaque outil, puis d’avoir une seule macro “diffuse” le même texte dans les deux chats en faisant simplement “coller + Enter”.

Dans VS Code, tu as déjà une base très propre parce que tu peux piloter beaucoup de choses via des command IDs et keybindings. Côté Codex, OpenAI documente des commandes VS Code (ex. `chatgpt.openSidebar`, `chatgpt.newChat`, etc.) que tu peux binder à un raccourci. Et côté chat “VS Code / Copilot”, il existe des commandes “workbench” pour ouvrir/pré-remplir le chat (`workbench.action.chat.open`) et même soumettre (`workbench.action.chat.submit`).

Antigravity, lui, est une application IDE séparée (multi-fenêtres, Agent Manager, etc.), donc la voie la plus universelle est l’automatisation desktop (UIA / accessibilité).

Concrètement, la solution “best bang for buck” sur Windows (vu ton environnement) : tu configures un raccourci dans VS Code qui met *à coup sûr* le curseur dans l’input du chat Codex, et **un raccourci dans Antigravity** qui met *à coup sûr* le curseur dans l’input de l’agent. Ensuite tu utilises **AutoHotkey v2** (ou Power Automate Desktop si tu veux 0 code) pour faire : activer VS Code → envoyer le raccourci “focus input” → coller → Enter → activer Antigravity → raccourci “focus input” → coller → Enter.

Voici un squelette AutoHotkey (tu adaptes juste les noms de fenêtres/process et les deux raccourcis “focus input”) :

```ahk
#Requires AutoHotkey v2.0

; --- CONFIG ---
VSCodeExe := "ahk_exe Code.exe"
AntiExe   := "ahk_exe Antigravity.exe"

; Ces deux hotkeys doivent exister dans chaque app et placer le caret dans la zone de prompt
VSCodeFocusHotkey := "^!1"   ; ex: Ctrl+Alt+1 = "focus chat Codex input" (à binder dans VS Code)
AntiFocusHotkey   := "^!2"   ; ex: Ctrl+Alt+2 = "focus agent input" (à binder dans Antigravity)

; --- UTIL ---
PastePreserve(text) {
    clipSaved := A_Clipboard
    A_Clipboard := text
    ClipWait 0.3
    Send "^v"
    Sleep 30
    A_Clipboard := clipSaved
}

; Hotkey global : Win+Enter envoie le presse-papier dans les 2 chats
#Enter:: {
    prompt := A_Clipboard
    if (prompt = "")
        return

    if WinExist(VSCodeExe) {
        WinActivate VSCodeExe
        WinWaitActive VSCodeExe, , 1
        Send VSCodeFocusHotkey
        Sleep 80
        PastePreserve(prompt)
        Send "{Enter}"
    }

    if WinExist(AntiExe) {
        WinActivate AntiExe
        WinWaitActive AntiExe, , 1
        Send AntiFocusHotkey
        Sleep 80
        PastePreserve(prompt)
        Send "{Enter}"
    }
}
```

src/extension.ts (extrait POC) :
```ts
import * as vscode from "vscode";
import * as http from "http";

function readJson(req: http.IncomingMessage): Promise<any> {
  return new Promise((resolve, reject) => {
    let data = "";
    req.on("data", chunk => (data += chunk));
    req.on("end", () => {
      try {
        resolve(data ? JSON.parse(data) : {});
      } catch (e) {
        reject(e);
      }
    });
    req.on("error", reject);
  });
}

export function activate(context: vscode.ExtensionContext) {
  const cfg = () => vscode.workspace.getConfiguration("promptBridge");
  const server = http.createServer(async (req, res) => {
    try {
      if (req.method !== "POST" || req.url !== "/send") {
        res.writeHead(404).end("not found");
        return;
      }

      const token = cfg().get<string>("token", "");
      const auth = req.headers["authorization"] || "";
      if (token && auth !== `Bearer ${token}`) {
        res.writeHead(401).end("unauthorized");
        return;
      }

      const body = await readJson(req);
      const prompt = String(body?.prompt || "").trim();
      const target = String(body?.target || "vscodeChat"); // "vscodeChat" | "codexBestEffort"
      if (!prompt) {
        res.writeHead(400).end("missing prompt");
        return;
      }

      const autoSend = cfg().get<boolean>("autoSend", true);

      if (target === "vscodeChat") {
        await vscode.commands.executeCommand("workbench.action.chat.open", {
          query: prompt,
          isPartialQuery: !autoSend
        });
      } else if (target === "codexBestEffort") {
        await vscode.commands.executeCommand("chatgpt.openSidebar");
        await vscode.commands.executeCommand("type", { text: prompt + (autoSend ? "\n" : "") });
      } else {
        res.writeHead(400).end("unknown target");
        return;
      }

      res.writeHead(200, { "content-type": "application/json" });
      res.end(JSON.stringify({ ok: true }));
    } catch (e: any) {
      res.writeHead(500).end(String(e?.message || e));
    }
  });

  const port = cfg().get<number>("port", 17373);
  server.listen(port, "127.0.0.1");

  context.subscriptions.push({ dispose: () => server.close() });

  context.subscriptions.push(
    vscode.commands.registerCommand("promptBridge.sendToChat", async () => {
      const prompt = await vscode.window.showInputBox({ prompt: "Prompt to send to Chat" });
      if (!prompt) return;
      const autoSend = cfg().get<boolean>("autoSend", true);
      await vscode.commands.executeCommand("workbench.action.chat.open", {
        query: prompt,
        isPartialQuery: !autoSend
      });
    })
  );
}

export function deactivate() {}
```

src/extension.ts :
```ts
import * as vscode from "vscode";
import * as http from "http";

function readJson(req: http.IncomingMessage): Promise<any> {
  return new Promise((resolve, reject) => {
    let data = "";
    req.on("data", (c) => (data += c));
    req.on("end", () => {
      try { resolve(data ? JSON.parse(data) : {}); }
      catch (e) { reject(e); }
    });
    req.on("error", reject);
  });
}

async function hasCommand(id: string): Promise<boolean> {
  const cmds = await vscode.commands.getCommands(true); // inclut internes
  return cmds.includes(id);
}

async function sendToAntigravity(prompt: string) {
  if (await hasCommand("antigravity.sendTextToChat")) {
    await vscode.commands.executeCommand("antigravity.sendTextToChat", true, prompt);
    return;
  }
  if (await hasCommand("antigravity.prioritized.chat.openNewConversation")) {
    await vscode.commands.executeCommand("antigravity.prioritized.chat.openNewConversation");
  }
  await vscode.commands.executeCommand("type", { text: prompt + "\n" });
}

async function sendToCodex(prompt: string) {
  const candidates = [
    "chatgpt.sendTextToChat",
    "chatgpt.sendMessage",
    "chatgpt.submit",
    "chatgpt.ask"
  ];
  for (const id of candidates) {
    if (await hasCommand(id)) {
      await vscode.commands.executeCommand(id, prompt);
      return;
    }
  }

  if (await hasCommand("chatgpt.openSidebar")) {
    await vscode.commands.executeCommand("chatgpt.openSidebar");
  }
  await vscode.env.clipboard.writeText(prompt);
  await vscode.commands.executeCommand("type", { text: prompt + "\n" });
}

export function activate(context: vscode.ExtensionContext) {
  const out = vscode.window.createOutputChannel("Prompt Bridge");
  out.appendLine("Prompt Bridge started");

  const cfg = vscode.workspace.getConfiguration("promptBridge");
  const port = cfg.get<number>("port", 17373);
  const token = cfg.get<string>("token", "");

  const server = http.createServer(async (req, res) => {
    try {
      if (req.method !== "POST" || req.url !== "/send") {
        res.writeHead(404).end("not found");
        return;
      }

      if (token) {
        const auth = req.headers["authorization"] || "";
        if (auth !== `Bearer ${token}`) {
          res.writeHead(401).end("unauthorized");
          return;
        }
      }

      const body = await readJson(req);
      const prompt = String(body?.prompt || "").trim();
      const target = String(body?.target || "codex"); // "codex" | "antigravity"
      if (!prompt) {
        res.writeHead(400).end("missing prompt");
        return;
      }

      out.appendLine(`Incoming target=${target}, chars=${prompt.length}`);

      if (target === "antigravity") await sendToAntigravity(prompt);
      else await sendToCodex(prompt);

      res.writeHead(200, { "content-type": "application/json" });
      res.end(JSON.stringify({ ok: true }));
    } catch (e: any) {
      out.appendLine("ERROR: " + String(e?.stack || e));
      res.writeHead(500).end(String(e?.message || e));
    }
  });

  server.listen(port, "127.0.0.1", () => {
    out.appendLine(`Listening on http://127.0.0.1:${port}/send`);
  });

  context.subscriptions.push({ dispose: () => server.close() });
}

export function deactivate() {}
```
